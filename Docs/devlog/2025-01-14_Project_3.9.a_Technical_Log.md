```python
@tool
async def spawn_and_log_sub_agent(
    parent_agent: str,
    sub_agent: str,
    mission: str,
    project_id: str
) -> dict:
    """Log sub-agent spawn event to database and broadcast via WebSocket"""
    try:
        # Create database record
        interaction = AgentInteraction(
            project_id=project_id,
            parent_agent=parent_agent,
            sub_agent=sub_agent,
            interaction_type="SPAWN",
            mission=mission
        )
        session.add(interaction)
        session.commit()
        
        # Broadcast WebSocket event
        await websocket_client.broadcast_sub_agent_spawned(
            parent_agent=parent_agent,
            sub_agent=sub_agent,
            mission=mission,
            project_id=project_id
        )
        
        return {
            "success": True,
            "interaction_id": str(interaction.id),
            "message": f"Sub-agent {sub_agent} spawned by {parent_agent}"
        }
    except Exception as e:
        logger.error(f"Failed to log sub-agent spawn: {e}")
        return {"success": False, "error": str(e)}
```

#### log_sub_agent_completion Tool
**File:** `src/giljo_mcp/tools/agent.py` (Lines 658-759)
```python
@tool
async def log_sub_agent_completion(
    parent_agent: str,
    sub_agent: str,
    result: str,
    duration_seconds: float,
    tokens_used: int,
    project_id: str,
    error: Optional[str] = None
) -> dict:
    """Log sub-agent completion event with metrics"""
    try:
        interaction_type = "ERROR" if error else "COMPLETE"
        
        # Update or create completion record
        interaction = session.query(AgentInteraction).filter_by(
            project_id=project_id,
            parent_agent=parent_agent,
            sub_agent=sub_agent,
            interaction_type="SPAWN"
        ).first()
        
        if interaction:
            # Update existing record
            interaction.interaction_type = interaction_type
            interaction.result = error or result
            interaction.duration_seconds = duration_seconds
            interaction.tokens_used = tokens_used
            interaction.updated_at = datetime.utcnow()
        else:
            # Create new completion record
            interaction = AgentInteraction(
                project_id=project_id,
                parent_agent=parent_agent,
                sub_agent=sub_agent,
                interaction_type=interaction_type,
                result=error or result,
                duration_seconds=duration_seconds,
                tokens_used=tokens_used
            )
            session.add(interaction)
        
        session.commit()
        
        # Broadcast WebSocket event
        await websocket_client.broadcast_sub_agent_completed(
            parent_agent=parent_agent,
            sub_agent=sub_agent,
            result=result,
            duration_seconds=duration_seconds,
            tokens_used=tokens_used,
            project_id=project_id,
            error=error
        )
        
        return {
            "success": True,
            "interaction_id": str(interaction.id),
            "type": interaction_type
        }
    except Exception as e:
        logger.error(f"Failed to log sub-agent completion: {e}")
        return {"success": False, "error": str(e)}
```

### WebSocket Event Broadcasting

```python
async def broadcast_sub_agent_spawned(self, parent_agent: str, sub_agent: str, 
                                      mission: str, project_id: str):
    """Broadcast sub-agent spawn event to all connected clients"""
    event = {
        "type": "sub_agent_spawned",
        "timestamp": datetime.utcnow().isoformat(),
        "data": {
            "parent_agent": parent_agent,
            "sub_agent": sub_agent,
            "mission": mission,
            "project_id": project_id
        }
    }
    await self.broadcast(json.dumps(event))

async def broadcast_sub_agent_completed(self, parent_agent: str, sub_agent: str,
                                        result: str, duration_seconds: float,
                                        tokens_used: int, project_id: str,
                                        error: Optional[str] = None):
    """Broadcast sub-agent completion event to all connected clients"""
    event = {
        "type": "sub_agent_completed" if not error else "sub_agent_error",
        "timestamp": datetime.utcnow().isoformat(),
        "data": {
            "parent_agent": parent_agent,
            "sub_agent": sub_agent,
            "result": result,
            "duration_seconds": duration_seconds,
            "tokens_used": tokens_used,
            "project_id": project_id,
            "error": error
        }
    }
    await self.broadcast(json.dumps(event))
```
