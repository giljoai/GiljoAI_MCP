ALL business logic goes in services. Endpoints are thin request/response
translators.

✅ CORRECT (endpoint uses service):

  @router.get("/")
  async def list_products(
      product_service: ProductService = Depends(get_product_service)
  ):
      # Endpoint only translates request → service → response

          raise HTTPException(status_code=400, detail=result["error"])
      return [ProductResponse.from_orm(p) for p in result["data"]]

  # src/giljo_mcp/services/product_service.py
  class ProductService:
      def __init__(self, session: AsyncSession, tenant_key: str):
          self.session = session
          self.tenant_key = tenant_key

      async def list_products(self) -> dict[str, Any]:
          """List all products for current tenant"""
          try:
              stmt = select(Product).where(
                  Product.tenant_key == self.tenant_key,
                  Product.is_deleted == False
              )
              result = await self.session.execute(stmt)
              products = result.scalars().all()
              return {"success": True, "data": products}
          except Exception as e:
              logger.error(f"Failed to list products: {e}")
              return {"success": False, "error": str(e)}

### Multi-Tenant Isolation (CRITICAL)

EVERY database query MUST filter by tenant_key. No exceptions.

❌ WRONG (no tenant isolation):
  stmt = select(Product).where(Product.id == product_id)  # SECURITY RISK

✅ CORRECT (tenant isolation enforced):
  stmt = select(Product).where(
      Product.id == product_id,
      Product.tenant_key == self.tenant_key  # REQUIRED
  )

Use BaseRepository pattern for automatic tenant filtering:

  # src/giljo_mcp/repositories/base_repository.py (lines 24-27)
  class BaseRepository:
      def __init__(self, session: AsyncSession, tenant_key: str):
          self.session = session
          self.tenant_key = tenant_key

      def _tenant_filter(self, model):
          """Automatically add tenant_key filter to all queries"""
          return model.tenant_key == self.tenant_key

Test multi-tenant isolation in every service test:

  // frontend/src/composables/useApiData.js
  import { ref } from 'vue'

  export function useApiData(endpoint) {
      const data = ref([])
      const loading = ref(false)
      const error = ref(null)

      async function fetch() {
          loading.value = true
          error.value = null
          try {
              const response = await api.get(endpoint)
              data.value = response.data
          } catch (e) {
              error.value = e.message
          } finally {
              loading.value = false
          }
      }

      return { data, loading, error, fetch }
  }

  // ProductsView.vue (uses composable)
  import { useApiData } from '@/composables/useApiData'

  export default {
      setup() {
          const { data: products, loading, fetch: fetchProducts } =
              useApiData('/products')

          onMounted(fetchProducts)

          return { products, loading, fetchProducts }
      }
  }

### Service Layer Pattern (Backend)

All services follow this structure:

  from sqlalchemy.ext.asyncio import AsyncSession
  from sqlalchemy import select
  from src.giljo_mcp.websocket_manager import websocket_manager
  import logging

  logger = logging.getLogger(__name__)

  class FeatureService:
      """Service for managing features with multi-tenant isolation"""

      def __init__(self, session: AsyncSession, tenant_key: str):
          self.session = session
          self.tenant_key = tenant_key

      async def create_feature(self, feature_data: dict) -> dict[str, Any]:
          """Create new feature

          Args:
              feature_data: Feature creation data

          Returns:
              dict with 'success' bool and 'data' or 'error'
          """
          try:
              feature = Feature(
                  name=feature_data["name"],
                  tenant_key=self.tenant_key
              )
              self.session.add(feature)
              await self.session.commit()

              # Emit WebSocket event
              await websocket_manager.broadcast_to_tenant(
                  tenant_key=self.tenant_key,
                  event="feature:created",
                  data={"feature_id": feature.id, "name": feature.name}
              )

              logger.info(
                  "Feature created",
                  extra={
                      "feature_id": feature.id,
                      "tenant_key": self.tenant_key
                  }
              )

              return {"success": True, "data": feature}

          except Exception as e:
              logger.error(
                  f"Failed to create feature: {e}",
                  extra={"tenant_key": self.tenant_key}
              )
              return {"success": False, "error": str(e)}

      async def list_features(self) -> dict[str, Any]:
          """List all features for current tenant"""
          try:
              stmt = select(Feature).where(
                  Feature.tenant_key == self.tenant_key,
                  Feature.is_deleted == False
              )
              result = await self.session.execute(stmt)
              features = result.scalars().all()

              return {"success": True, "data": features}

          except Exception as e:
              logger.error(f"Failed to list features: {e}")
              return {"success": False, "error": str(e)}

### Multi-Tenant Isolation Pattern

EVERY query MUST filter by tenant_key. Use this pattern:

  async def get_feature(self, feature_id: int) -> dict[str, Any]:
      """Get feature by ID (tenant-isolated)"""
      try:
          stmt = select(Feature).where(
              Feature.id == feature_id,
              Feature.tenant_key == self.tenant_key  # REQUIRED
          )
          result = await self.session.execute(stmt)
          feature = result.scalar_one_or_none()

          if not feature:
              return {
                  "success": False,
                  "error": f"Feature {feature_id} not found"
              }

          return {"success": True, "data": feature}

      except Exception as e:
          return {"success": False, "error": str(e)}

### Frontend Composable Pattern (Vue 3)

Extract reusable logic into composables:

  // frontend/src/composables/useFeatureData.js
  import { ref, computed } from 'vue'
  import api from '@/services/api'

  export function useFeatureData() {
      const features = ref([])
      const loading = ref(false)
      const error = ref(null)

      const activeFeatures = computed(() =>
          features.value.filter(f => f.status === 'active')
      )

      async function fetchFeatures() {
          loading.value = true
          error.value = null
          try {
              const response = await api.get('/features')
              features.value = response.data
          } catch (e) {
              error.value = e.message
          } finally {
              loading.value = false
          }
      }

      async function createFeature(featureData) {
          try {
              const response = await api.post('/features', featureData)
              features.value.push(response.data)
              return { success: true, data: response.data }
          } catch (e) {
              return { success: false, error: e.message }
          }
      }

      return {
          features,
          loading,
          error,
          activeFeatures,
          fetchFeatures,
          createFeature
      }
  }

## 10. SCOPE MANAGEMENT & ANTI-PATTERNS
══════════════════════════════════════════════════════════════════════════════

GiljoAI has experienced scope creep in handover series. Learn from history.

### Case Study: 0240-0243 Series (2 weeks → 6 weeks)

**Original Plan** (0240 series):
- 4 handovers (0240a-d)
- 24-30 hours effort
- 3-4 days wall-clock time
- GUI redesign matching PDF vision

**What Went Wrong**:
- Used wrong PDF template (missing reference screenshots)
- 0240a-d completed but didn't match actual design
- Emergency fix 0241 (complete rewrite from scratch)
- Refinement attempt 0242 (scope gaps, abandoned)
- Final success 0243 (Nicepage conversion, 6 handovers)

**Timeline Impact**:
- Original estimate: 2 weeks
- Actual completion: 6+ weeks (300% overrun)
- Total handovers: 4 planned → 18 created (superseded + final)

**Lessons Learned**:

1. **VERIFY REQUIREMENTS FIRST**
   - Before implementing, verify EXACT requirements
   - Compare screenshots/mockups to PDF vision
   - Ask user to confirm design references
   - Don't assume "this looks right"

2. **SMALL HANDOVERS, FAST VALIDATION**
   - Better: 6 small handovers (0243a-f) with intermediate validation
   - Worse: 4 large handovers with late discovery of wrong design

3. **FAIL FAST, PIVOT QUICKLY**
   - 0240 series completed but wrong → immediate 0241 pivot
   - Don't "fix" a fundamentally wrong approach
   - Sometimes complete rewrite is faster than incremental fixes

4. **AVOID PARALYSIS BY PLANNING**
   - 0240_series_execution_plan.md was 580 lines
   - Detailed plans don't prevent wrong implementation
   - Execution + validation > elaborate planning

5. **HANDOVER RETIREMENT IS OK**
   - Archive superseded handovers clearly
   - Keep for historical reference (learn from mistakes)
   - Don't delete (git history isn't enough context)

### Anti-Pattern Checklist

Before starting a handover series, ask:

❌ **Anti-Pattern**: "Let's plan everything in detail first"
✅ **Best Practice**: "Let's implement Phase 1, validate, then plan Phase 2"

❌ **Anti-Pattern**: "This design looks close enough to the vision"
✅ **Best Practice**: "Show me side-by-side screenshots - exact match required"

❌ **Anti-Pattern**: "We can refine this incrementally"
✅ **Best Practice**: "Is this the right foundation? Or do we need to pivot?"

❌ **Anti-Pattern**: "Let's fix all the edge cases in one handover"
✅ **Best Practice**: "Critical path first, edge cases in follow-up handovers"

❌ **Anti-Pattern**: "This handover should cover everything"
✅ **Best Practice**: "This handover has ONE clear goal with validation"

══════════════════════════════════════════════════════════════════════════════

### When to Use This Approach

Apply this methodology when:
✓ Removing deprecated/legacy functionality (agent coordination, old APIs)

✓ Deleting core modules that have widespread imports/usage
✓ Refactoring that touches 5+ files across multiple layers

### Pre-Refactor Checklist

Before deleting ANYTHING:

□ CASCADE IMPACT ANALYSIS COMPLETE

  - Phase 2: Delete legacy files
  - Phase 3: Fix cascading imports
  - Phase 4: Verification and cleanup

□ STAKEHOLDER ALIGNMENT
  - User confirms legacy code is no longer needed
  - Team agrees on replacement approach
  - Timeline expectations set (this takes multiple handovers)

### Anti-Patterns to Avoid During Refactors

❌ **WRONG: Delete everything at once without cascade analysis**
  # This causes chaos - unknown impact, random test failures
  rm -rf src/giljo_mcp/legacy/
  pytest tests/  # 157 failures - no idea which are expected

✅ **CORRECT: Cascade analysis FIRST, then phased deletion**
  # Document every impacted file before deletion
  # Create Expected Failures list
  # Delete in Phase 2, fix in Phase 3

❌ **WRONG: Comment out legacy code "just in case"**
  # from agent_job_manager import AgentJobManager  # OLD
  # Use new pattern instead
  from orchestration_service import cancel_job

✅ **CORRECT: Delete completely, rely on git history**
  from orchestration_service import cancel_job
  # Git history: git log -p -- src/file.py (if needed)

❌ **WRONG: Fix cascades randomly as you encounter them**
  # This creates inconsistent state - some layers use new, some old
  # Fix api/endpoints/tasks.py
  # Oops, forgot services still use old pattern
  # Now API and services are inconsistent

✅ **CORRECT: Fix cascades layer-by-layer (bottom-up)**
  # LAYER 1: Models (foundation)
  # LAYER 2: Services (business logic)
  # LAYER 3: Tools (MCP tools)
  # LAYER 4: API (endpoints)
  # LAYER 5: Frontend (if needed)
  # Each layer builds on previous layer's fixes

❌ **WRONG: Skip verification phase "tests pass, we're done"**
  pytest tests/ -v  # All pass
  git commit -m "refactor complete"
  # 3 months later: "Why is this commented code here?"

✅ **CORRECT: Thorough verification with orphan detection**
  grep -r "LegacyModule" src/  # Should be ZERO matches
  grep -r "# from.*LegacyModule" src/  # No zombie imports
  pytest tests/ --cov=src/giljo_mcp --cov-report=html
  # Manual testing of critical workflows

❌ **WRONG: No documentation of what was removed**
  # Future developers: "Why don't we have AgentJobManager anymore?"
  # No historical context, might re-implement same pattern

✅ **CORRECT: Document refactor decisions in handover**
  # handovers/0420_series_summary.md
  - What: Removed AgentJobManager, AgentCommunicationQueue
  - Why: Functionality moved to OrchestrationService (service layer)
  - Replacement: cancel_job() MCP tool + service methods
  - Impact: 29 files across 5 layers
  - Timeline: 4 handovers (0420a-d), 2 weeks

  async def cancel_job(job_id: str, tenant_key: str) -> dict:
      """Cancel active agent job"""
      async with get_db_session() as session:
          stmt = select(AgentJob).where(
              AgentJob.id == job_id,
              AgentJob.tenant_key == tenant_key
          )
          result = await session.execute(stmt)
          job = result.scalar_one_or_none()

          if not job:
              return {"success": False, "error": "Job not found"}

          job.status = "cancelled"
          await session.commit()

          return {"success": True, "data": {"job_id": job_id}}
