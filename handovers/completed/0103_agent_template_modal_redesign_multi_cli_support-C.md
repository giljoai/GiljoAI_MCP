## Overview

Redesign the "Create Agent" and "Edit Agent" modals in the Agent Template Manager to:
1. Support multiple CLI tools (Claude Code, Codex CLI, Gemini CLI, Generic)
2. Align with [Claude Code's subagent format](https://docs.claude.com/en/docs/claude-code/sub-agents#cli-based-configuration)
3. Auto-generate agent names from Role + Custom Suffix
4. Simplify UX by removing unclear fields (variables, augmentation_slots)
5. Add preview/summary window with copyable generated content
6. Maintain GiljoAI's existing role-based color system

```python
"""Agent template validation functions for 0103."""
import re
from typing import Tuple
from sqlalchemy.orm import Session
from src.giljo_mcp.models import AgentTemplate


def slugify_name(role: str, suffix: str = None) -> str:
    """Generate agent name from role and optional suffix.

    Args:
        role: Agent role (e.g., 'orchestrator')
        suffix: Optional custom suffix (e.g., 'AmazingGuy')

    Returns:
        Slugified name (e.g., 'orchestrator-amazing-guy')
    """
    if suffix:
        # Convert to lowercase and replace spaces/underscores with hyphens
        suffix_clean = re.sub(r'[^a-z0-9-]', '', suffix.lower().replace('_', '-').replace(' ', '-'))
        return f"{role}-{suffix_clean}"
    return role


def validate_agent_name(name: str, tenant_key: str, db: Session, exclude_id: str = None) -> Tuple[bool, str]:
    """Validate agent name format and uniqueness.

    Args:
        name: Proposed agent name
        tenant_key: Tenant isolation key
        db: Database session
        exclude_id: Template ID to exclude from uniqueness check (for updates)

    Returns:
        Tuple of (is_valid, error_message)
    """
    # Check format: lowercase letters, numbers, hyphens only
    if not re.match(r'^[a-z0-9]+(-[a-z0-9]+)*$', name):
        return False, "Name must use lowercase letters, numbers, and hyphens only"

    # Check length
    if len(name) > 100:
        return False, "Name must be 100 characters or less"

    # Check uniqueness within tenant
    query = db.query(AgentTemplate).filter_by(
        tenant_key=tenant_key,
        name=name
    )

    if exclude_id:
        query = query.filter(AgentTemplate.id != exclude_id)

    existing = query.first()
    if existing:
        return False, f"Agent name '{name}' already exists"

    return True, ""


def validate_system_prompt(content: str) -> Tuple[bool, str]:
    """Validate system prompt content.

    Args:
        content: System prompt text

    Returns:
        Tuple of (is_valid, error_message)
    """
    if not content or not content.strip():
        return False, "System prompt is required"

    if len(content.strip()) < 20:
        return False, "System prompt is too short (minimum 20 characters)"

    return True, ""


def can_activate_role(role: str, tenant_key: str, db: Session, exclude_id: str = None) -> Tuple[bool, str]:
    """Check if activating this role would exceed 8-role limit.

    Args:
        role: Role to activate
        tenant_key: Tenant isolation key
        db: Database session
        exclude_id: Template ID to exclude from count (for updates)

    Returns:
        Tuple of (can_activate, error_message)
    """
    # Count currently active distinct roles
    query = db.query(AgentTemplate.role).filter_by(
        tenant_key=tenant_key,
        is_active=True
    ).distinct()

    if exclude_id:
        query = query.filter(AgentTemplate.id != exclude_id)

    active_roles = {row[0] for row in query.all()}

    # If this role is already active elsewhere, allow toggle
    if role in active_roles:
        return True, ""

    # If we have 8 distinct active roles, block new role activation
    if len(active_roles) >= 8:
        return False, f"Maximum 8 active agent roles allowed (currently {len(active_roles)}). Deactivate another role first."

    return True, ""


def get_role_color(role: str) -> str:
    """Get background color for a role.

    Args:
        role: Agent role

    Returns:
        Hex color code
    """
    color_map = {
        'orchestrator': '#D4A574',
        'analyzer': '#E74C3C',
        'designer': '#9B59B6',
        'frontend': '#3498DB',
        'backend': '#2ECC71',
        'implementer': '#3498DB',
        'tester': '#FFC300',
        'reviewer': '#9B59B6',
        'documenter': '#27AE60',
    }
    return color_map.get(role, '#90A4AE')
```

```python
"""Agent template rendering to CLI tool formats (0103)."""
import yaml
from typing import Dict, Any
from src.giljo_mcp.models import AgentTemplate


def render_claude_agent(template: AgentTemplate) -> str:
    """Render agent template to Claude Code YAML format.

    Per Claude Code spec: https://docs.claude.com/en/docs/claude-code/sub-agents

    Args:
        template: AgentTemplate model instance

    Returns:
        Markdown string with YAML frontmatter + body
    """
    # Build frontmatter
    frontmatter = {
        'name': template.name,
        'description': template.description or f'Subagent for {template.role}',
    }

    # Add model (default to sonnet if blank)
    model = template.model or 'sonnet'
    frontmatter['model'] = model

    # IMPORTANT: Omit tools field to inherit all (per Claude Code spec)
    # Do NOT add: frontmatter['tools'] = ...

    # Build body parts
    body_parts = [template.template_content]

    # Add behavioral rules section if present
    if template.behavioral_rules:
        body_parts.append('\n## Behavioral Rules')
        for rule in template.behavioral_rules:
            body_parts.append(f'- {rule}')

    # Add success criteria section if present
    if template.success_criteria:
        body_parts.append('\n## Success Criteria')
        for criterion in template.success_criteria:
            body_parts.append(f'- {criterion}')

    body = '\n'.join(body_parts)

    # Assemble YAML + body
    yaml_header = yaml.dump(frontmatter, default_flow_style=False, sort_keys=False)
    return f'---\n{yaml_header}---\n\n{body}'


def render_generic_agent(template: AgentTemplate) -> str:
    """Render agent template to generic plaintext format.

    Args:
        template: AgentTemplate model instance

    Returns:
        Plaintext prompt
    """
    parts = [
        f"# {template.name}",
        f"\nRole: {template.role}",
        f"\n{template.template_content}",
    ]

    if template.behavioral_rules:
        parts.append('\n## Behavioral Rules')
        for rule in template.behavioral_rules:
            parts.append(f'- {rule}')

    if template.success_criteria:
        parts.append('\n## Success Criteria')
        for criterion in template.success_criteria:
            parts.append(f'- {criterion}')

    return '\n'.join(parts)


def render_template(template: AgentTemplate) -> str:
    """Render template based on cli_tool.

    Args:
        template: AgentTemplate model instance

    Returns:
        Rendered content (YAML for Claude, plaintext for others)
    """
    if template.cli_tool == 'claude':
        return render_claude_agent(template)
    elif template.cli_tool in ('codex', 'gemini', 'generic'):
        return render_generic_agent(template)
    else:
        # Fallback
        return render_claude_agent(template)
```

```python
from src.giljo_mcp.template_validation import (
    slugify_name,
    validate_agent_name,
    validate_system_prompt,

@router.post("/", response_model=TemplateResponse)
async def create_template(
    template: TemplateCreate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db_session)
):
    """Create new agent template."""

    # Generate name from role + suffix
    if template.custom_suffix:
        generated_name = slugify_name(template.role, template.custom_suffix)
    else:
        generated_name = template.name or template.role

    # Validate name
    is_valid, error_msg = validate_agent_name(
        generated_name,
        current_user.tenant_key,
        db
    )
    if not is_valid:
        raise HTTPException(status_code=400, detail=error_msg)

    # Validate system prompt
    is_valid, error_msg = validate_system_prompt(template.template_content)
    if not is_valid:
        raise HTTPException(status_code=400, detail=error_msg)

    # Auto-assign background color if not provided
    background_color = template.background_color or get_role_color(template.role)

    # Set default description if not provided (Claude only)
    description = template.description
    if not description and template.cli_tool == 'claude':
        description = f'Subagent for {template.role}'

    # Create template
    new_template = AgentTemplate(
        id=str(uuid4()),
        tenant_key=current_user.tenant_key,
        name=generated_name,
        role=template.role,
        cli_tool=template.cli_tool,
        background_color=background_color,
        description=description,
        template_content=template.template_content,
        model=template.model or 'sonnet',
        tools=None,  # Always inherit all
        behavioral_rules=template.behavioral_rules,
        success_criteria=template.success_criteria,
        tags=template.tags,
        is_default=template.is_default,
        is_active=template.is_active,
    )

    db.add(new_template)
    await db.commit()
    await db.refresh(new_template)

    return new_template
```

```python
from src.giljo_mcp.template_validation import can_activate_role

@router.patch("/{template_id}", response_model=TemplateResponse)
async def update_template(
    template_id: str,
    update: TemplateUpdate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db_session)
):
    """Update agent template."""

    template = await db.get(AgentTemplate, template_id)

    if not template or template.tenant_key != current_user.tenant_key:
        raise HTTPException(status_code=404, detail="Template not found")

    # If toggling is_active to True, check 8-role limit
    if update.is_active is True and not template.is_active:
        can_activate, error_msg = can_activate_role(
            template.role,
            current_user.tenant_key,
            db,
            exclude_id=template_id
        )
        if not can_activate:
            raise HTTPException(status_code=409, detail=error_msg)

    # Apply updates
    if update.name is not None:
        # Validate new name
        is_valid, error_msg = validate_agent_name(
            update.name,
            current_user.tenant_key,
            db,
            exclude_id=template_id
        )
        if not is_valid:
            raise HTTPException(status_code=400, detail=error_msg)
        template.name = update.name

    if update.role is not None:
        template.role = update.role
        # Auto-update background_color when role changes
        template.background_color = get_role_color(update.role)

    if update.cli_tool is not None:
        template.cli_tool = update.cli_tool

    if update.background_color is not None:
        template.background_color = update.background_color

    if update.description is not None:
        template.description = update.description

    if update.template_content is not None:
        is_valid, error_msg = validate_system_prompt(update.template_content)
        if not is_valid:
            raise HTTPException(status_code=400, detail=error_msg)
        template.template_content = update.template_content

    if update.model is not None:
        template.model = update.model

    if update.behavioral_rules is not None:
        template.behavioral_rules = update.behavioral_rules

    if update.success_criteria is not None:
        template.success_criteria = update.success_criteria

    if update.tags is not None:
        template.tags = update.tags

    if update.is_default is not None:
        template.is_active = update.is_active

    template.updated_at = datetime.utcnow()

    await db.commit()
    await db.refresh(template)

    return template
```

```python
def render_claude_agent(template):
    """Render agent template to Claude Code format."""

    # Build frontmatter
    frontmatter = {
        'name': template.name,
        'description': template.description or f'Subagent for {template.role}',
    }

    # Add model (default to sonnet if blank)
    model = template.model or 'sonnet'
    frontmatter['model'] = model

    # Omit tools field to inherit all (per Claude Code spec)
    # Do NOT add: frontmatter['tools'] = ...

    # Build body
    body_parts = [template.template_content]

    # Add behavioral rules section if present
    if template.behavioral_rules:
        body_parts.append('\n## Behavioral Rules')
        for rule in template.behavioral_rules:
            body_parts.append(f'- {rule}')

    # Add success criteria section if present
    if template.success_criteria:
        body_parts.append('\n## Success Criteria')
        for criterion in template.success_criteria:
            body_parts.append(f'- {criterion}')

    body = '\n'.join(body_parts)

    # Assemble YAML + body
    yaml_header = yaml.dump(frontmatter, default_flow_style=False)
    return f'---\n{yaml_header}---\n\n{body}'
```

### Name Validation

```python
import re

def validate_agent_name(name: str, tenant_key: str) -> tuple[bool, str]:
    """Validate agent name format and uniqueness."""

    # Must be lowercase with hyphens only
    if not re.match(r'^[a-z0-9]+(-[a-z0-9]+)*$', name):
        return False, "Name must use lowercase letters, numbers, and hyphens only"

    # Check uniqueness within tenant
    existing = db.query(AgentTemplate).filter_by(
        tenant_key=tenant_key,
        name=name
    ).first()

    if existing:
        return False, f"Agent name '{name}' already exists"

    return True, ""
```

```python
        return False, "System prompt is required"

    if len(content.strip()) < 20:
        return False, "System prompt is too short (minimum 20 characters)"

    return True, ""
```

### Active Role Limit (8 Max)

```python
def can_activate_role(role: str, tenant_key: str, exclude_id: str = None) -> tuple[bool, str]:
    """Check if activating this role would exceed 8-role limit."""

    # Count currently active distinct roles
    query = db.query(AgentTemplate.role).filter_by(
        tenant_key=tenant_key,
        is_active=True
    )

    if exclude_id:
        query = query.filter(AgentTemplate.id != exclude_id)

    active_roles = set(query.distinct().all())

    # If this role is already active elsewhere, allow toggle
    if role in active_roles:
        return True, ""

    # If we have 8 distinct active roles, block new role activation
    if len(active_roles) >= 8:
        return False, f"Maximum 8 active agent roles allowed (currently {len(active_roles)}). Deactivate another role first."

    return True, ""
```
