     """
     Database models for TinyContacts application
     """
     from datetime import datetime
     from app import db


     class Contact(db.Model):
         """Contact model for storing contact information"""

         __tablename__ = 'contacts'

         # Primary key
         id = db.Column(db.Integer, primary_key=True)

         # Basic information
         first_name = db.Column(db.String(50), nullable=False)
         last_name = db.Column(db.String(50), nullable=False)
         email = db.Column(db.String(120), unique=True, nullable=True)
         phone = db.Column(db.String(20), nullable=True)

         # Additional information
         company = db.Column(db.String(100), nullable=True)
         job_title = db.Column(db.String(100), nullable=True)
         notes = db.Column(db.Text, nullable=True)

         # Profile picture
         profile_picture = db.Column(db.String(200), nullable=True)  # File path

         # Timestamps
         created_at = db.Column(db.DateTime, default=datetime.utcnow)
         updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

         def __repr__(self):
             return f'<Contact {self.first_name} {self.last_name}>'

         @property
         def full_name(self):
             """Return the full name of the contact"""
             return f"{self.first_name} {self.last_name}"

         def to_dict(self):
             """Convert contact to dictionary for JSON serialization"""
             return {
                 'id': self.id,
                 'first_name': self.first_name,
                 'last_name': self.last_name,
                 'email': self.email,
                 'phone': self.phone,
                 'company': self.company,
                 'job_title': self.job_title,
                 'notes': self.notes,
                 'profile_picture': self.profile_picture,
                 'created_at': self.created_at.isoformat() if self.created_at else None,
                 'updated_at': self.updated_at.isoformat() if self.updated_at else None
             }

     # TinyContacts

     A lightweight, modern contact management application built with Flask and SQLAlchemy.

     ## Overview

     TinyContacts is a simple yet powerful contact management system that allows users to store, organize, and manage
     their personal and professional contacts. The application focuses on essential contact management features with a
     clean, intuitive interface.

     ## Features

     ### Core Contact Management
     - **CRUD Operations**: Create, read, update, and delete contacts
     - **Contact Information**: Store names, phone numbers, emails, addresses, and notes
     - **Photo Upload**: Add and manage contact photos
     - **Date Management**: Track important dates like birthdays and anniversaries
     - **Search and Filter**: Quickly find contacts using various criteria

     ### User Experience
     - **Responsive Design**: Works seamlessly across desktop and mobile devices
     - **Clean Interface**: Minimalist design focused on usability
     - **Quick Actions**: One-click access to common operations
     - **Data Export**: Export contact data for backup and portability

     ## Technology Stack

     ### Backend
     - **Flask**: Lightweight Python web framework
     - **SQLAlchemy**: Object-relational mapping (ORM)
     - **Flask-Migrate**: Database migration management
     - **Flask-WTF**: Form handling and validation
     - **Werkzeug**: File upload handling

     ### Frontend
     - **Jinja2**: Template engine for dynamic HTML
     - **Bootstrap**: Responsive CSS framework
     - **JavaScript**: Interactive frontend features

     ### Database
     - **SQLite**: Default development database
     - **PostgreSQL**: Production database option

     ### File Storage
     - **Local Storage**: Photo and file management
     - **Configurable Paths**: Customizable upload directories

         company = db.Column(db.String(100), nullable=True)

         # Profile picture

         # Timestamps
         created_at = db.Column(db.DateTime, default=datetime.utcnow)
         updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

         def __repr__(self):
             return f'<Contact {self.first_name} {self.last_name}>'

         @property
         def full_name(self):

     ## Overview

     TinyContacts follows a traditional Model-View-Controller (MVC) architecture pattern built on top of the Flask web
     framework. The application is designed to be lightweight, maintainable, and easily extensible while providing
     robust contact management functionality.

     ### Design Principles

     - **Simplicity**: Keep the codebase clean and understandable
     - **Modularity**: Separate concerns with clear boundaries
     - **Scalability**: Design for future growth and feature additions
     - **Security**: Implement secure coding practices from the ground up
     - **Performance**: Optimize for fast response times and efficient resource usage

     ## Technology Stack

     ### Backend Framework
     - **Flask 3.0.0**: Lightweight Python web framework
       - **Rationale**: Simple to learn, flexible, and perfect for small to medium applications
       - **Benefits**: Minimal boilerplate, extensive ecosystem, active community

     ### Database Layer
     - **SQLAlchemy 2.0.23**: Object-relational mapping (ORM)
       - **Rationale**: Powerful ORM with excellent Flask integration
       - **Benefits**: Database abstraction, migration support, relationship management
     - **Flask-SQLAlchemy 3.1.1**: Flask integration for SQLAlchemy
       - **Benefits**: Simplified configuration and session management

     ### Database Options
     - **SQLite**: Default development database
       - **Use Case**: Local development, testing, small deployments
       - **Benefits**: No setup required, file-based, lightweight
     - **PostgreSQL**: Recommended production database
       - **Use Case**: Production deployments, high concurrency
       - **Benefits**: ACID compliance, advanced features, scalability

     ### Form Handling
     - **Flask-WTF 1.2.1**: Form handling and validation
       - **Benefits**: CSRF protection, easy validation, Flask integration
     - **WTForms 3.1.0**: Form definition and validation library
       - **Benefits**: Flexible validation, multiple field types, custom validators

     ### Image Processing
     - **Pillow 10.1.0**: Image processing library
       - **Use Case**: Profile picture processing and optimization
       - **Benefits**: Format support, resizing, thumbnail generation

     ### Configuration Management
     - **python-dotenv 1.0.0**: Environment variable management
       - **Benefits**: Secure configuration, environment separation

     ### Development Tools
     - **pytest 7.4.3**: Testing framework
     - **pytest-flask 1.3.0**: Flask-specific testing utilities
     - **black 23.11.0**: Code formatting
     - **flake8 6.1.0**: Code linting

     ## Database Design

     ### Entity Relationship Diagram

     ```
     ┌─────────────────────────────────────────┐
     │                CONTACT                  │
     ├─────────────────────────────────────────┤
     │ PK │ id          │ INTEGER             │
     │    │ first_name  │ VARCHAR(50)  [NN]   │
     │    │ last_name   │ VARCHAR(50)  [NN]   │
     │    │ email       │ VARCHAR(120) [U]    │
     │    │ phone       │ VARCHAR(20)         │
     │    │ company     │ VARCHAR(100)        │
     │    │ job_title   │ VARCHAR(100)        │
     │    │ notes       │ TEXT                │
     │    │ profile_pic │ VARCHAR(200)        │
     │    │ created_at  │ DATETIME    [NN]    │
     │    │ updated_at  │ DATETIME    [NN]    │
     └─────────────────────────────────────────┘

     Legend:
     PK = Primary Key
     NN = Not Null
     U  = Unique
     ```

     ### Contact Model Schema

     #### Core Fields
     - **id**: Primary key, auto-incrementing integer
     - **first_name**: Required, up to 50 characters
     - **last_name**: Required, up to 50 characters

     #### Contact Information
     - **email**: Optional, unique, up to 120 characters
     - **phone**: Optional, up to 20 characters (supports various formats)

     #### Professional Information
     - **company**: Optional company name, up to 100 characters
     - **job_title**: Optional position title, up to 100 characters

     #### Additional Data
     - **notes**: Optional text field for free-form information
     - **profile_picture**: Optional file path to profile image

     #### Metadata
     - **created_at**: Automatic timestamp on record creation
     - **updated_at**: Automatic timestamp on record modification

     ### Database Constraints

     #### Business Rules
     1. **Name Requirements**: Both first_name and last_name are required
     2. **Email Uniqueness**: Email addresses must be unique when provided
     3. **Data Integrity**: All timestamps are automatically managed
     4. **File References**: Profile pictures stored as file paths, not binary data

     #### Indexing Strategy
     - **Primary Index**: id (automatic)
     - **Unique Index**: email (automatic with UNIQUE constraint)
     - **Future Indexes**:
       - Composite index on (last_name, first_name) for name-based searches
       - Index on company for organization-based queries

     ### Migration Strategy
     - **Flask-Migrate**: For database schema evolution
     - **Version Control**: All migrations tracked in version control
     - **Rollback Support**: Reversible migrations for safe deployment

     ## Troubleshooting

     ### Common Issues and Solutions

     #### Issue: ModuleNotFoundError
     **Problem**: Python can't find installed packages
     **Solution**:
     ```bash
     # Ensure virtual environment is activated
     source venv/bin/activate  # or venv\Scripts\activate on Windows

     # Verify Python path
     which python
     python -c "import sys; print(sys.path)"

     # Reinstall requirements if needed
     pip install -r requirements.txt
     ```

     #### Issue: Database Connection Error
     **Problem**: Can't connect to database
     **Solutions**:

     For SQLite:
     ```bash
     # Check file permissions
     ls -la contacts.db
     # Ensure directory is writable
     touch test_file && rm test_file
     ```

     For PostgreSQL:
     ```bash
     # Test connection
     psql -h localhost -U tinycontacts_user -d tinycontacts

     # Check service status
     sudo systemctl status postgresql
     ```

     #### Issue: File Upload Errors
     **Problem**: Can't upload profile pictures
     **Solution**:
     ```bash
     # Create uploads directory
     mkdir uploads

     # Check permissions (Linux/macOS)
     chmod 755 uploads

     # Check disk space
     df -h .
     ```

     #### Issue: Port Already in Use
     **Problem**: Address already in use error
     **Solutions**:
     ```bash
     # Find process using port 5000
     lsof -i :5000  # macOS/Linux
     netstat -ano | findstr :5000  # Windows

     # Kill process if needed
     kill -9 <PID>  # macOS/Linux

     # Or use different port
     python src/app.py --port 8000
     ```

     #### Issue: Import Errors in app.py
     **Problem**: Cannot import from models
     **Solution**:
     ```bash
     # Ensure you're running from project root
     pwd  # Should show TinyContacts directory

     # Run with Python module syntax
     python -m src.app

     # Or add src to Python path
     export PYTHONPATH="${PYTHONPATH}:./src"
     python src/app.py
     ```

     ## Overview

     The TinyContacts API provides RESTful endpoints for managing contacts programmatically. The API follows REST
     conventions and returns JSON responses for all operations.

     ### API Features
     - **Full CRUD Operations**: Create, read, update, and delete contacts
     - **Search and Filtering**: Find contacts by various criteria
     - **Pagination**: Handle large contact lists efficiently
     - **File Upload**: Manage profile pictures
     - **Data Export**: Retrieve contact data in various formats

     ### Current Status
     ⚠️ **Note**: The API is currently under development. Endpoints marked as "Planned" will be implemented in future
     releases.

     ## Error Handling

     ### HTTP Status Codes
     | Code | Description | When Used |
     |------|-------------|-----------|
     | 200 | OK | Successful GET, PUT requests |
     | 201 | Created | Successful POST requests |
     | 204 | No Content | Successful DELETE requests |
     | 400 | Bad Request | Invalid request data |
     | 401 | Unauthorized | Authentication required |
     | 403 | Forbidden | Access denied |
     | 404 | Not Found | Resource doesn't exist |
     | 409 | Conflict | Duplicate email address |
     | 422 | Unprocessable Entity | Validation errors |
     | 500 | Internal Server Error | Server error |

     ### Error Response Structure
     ```json
     {
       "status": "error",
       "error": {
         "code": "ERROR_CODE",
         "message": "Human-readable error message",
         "details": {
           "field_name": ["Field-specific error messages"]
         }
       }
     }
     ```

     ### Common Error Codes
     | Code | Description | HTTP Status |
     |------|-------------|-------------|
     | `VALIDATION_ERROR` | Input validation failed | 422 |
     | `NOT_FOUND` | Resource not found | 404 |
     | `DUPLICATE_EMAIL` | Email already exists | 409 |
     | `INVALID_FORMAT` | Invalid data format | 400 |
     | `FILE_TOO_LARGE` | Uploaded file exceeds limit | 413 |

     import requests
     import json

     # Base URL

     # Create a new contact

         contact_id = contact["id"]

         # Get the contact
         response = requests.get(f"{base_url}/contacts/{contact_id}")
         if response.status_code == 200:
             contact = response.json()["data"]["contact"]
             print(f"Retrieved: {contact}")

         # Update the contact
         update_data = {"job_title": "Senior Developer"}
         response = requests.put(
             f"{base_url}/contacts/{contact_id}",
             headers={"Content-Type": "application/json"},
             data=json.dumps(update_data)
         )

         # List all contacts
         response = requests.get(f"{base_url}/contacts?per_page=5")
         if response.status_code == 200:
             data = response.json()["data"]
             print(f"Found {data['pagination']['total']} contacts")

     else:
         print(f"Error: {response.status_code}")
         print(response.json())

     ### JavaScript Example (using fetch)
     ```javascript
     const baseUrl = 'http://localhost:5000/api/v1';

     // Create a new contact
     async function createContact() {
       const contactData = {
         first_name: 'Bob',
         last_name: 'Wilson',
         email: 'bob.wilson@example.com',
         phone: '555-0126'
       };

       try {
         const response = await fetch(`${baseUrl}/contacts`, {
           method: 'POST',
           headers: {
             'Content-Type': 'application/json',
           },
           body: JSON.stringify(contactData)
         });

         if (response.ok) {
           const result = await response.json();
           const contact = result.data.contact;
           console.log(`Created contact: ${contact.first_name} ${contact.last_name}`);
           return contact.id;
         } else {
           const error = await response.json();
           console.error('Error:', error);
         }
       } catch (error) {
         console.error('Network error:', error);
       }
     }

     // Search contacts
     async function searchContacts(searchTerm) {
       try {
         const response = await fetch(
           `${baseUrl}/contacts?search=${encodeURIComponent(searchTerm)}&per_page=10`
         );

         if (response.ok) {
           const result = await response.json();
           const contacts = result.data.contacts;
           console.log(`Found ${contacts.length} contacts matching "${searchTerm}"`);
           contacts.forEach(contact => {
             console.log(`- ${contact.first_name} ${contact.last_name} (${contact.email})`);
           });
         }
       } catch (error) {
         console.error('Search error:', error);
       }
     }

     // Usage
     createContact().then(contactId => {
       if (contactId) {
         console.log(`Contact created with ID: ${contactId}`);
       }
     });

     searchContacts('wilson');
     ```

     #### Token Generation Flow

     **Step 1: Generate Token** (POST /api/download/generate-token)
     File: api/endpoints/downloads.py (lines 469-593)

     1. User authenticates (JWT cookie or X-API-Key header)
     2. Request body: {"content_type": "agent_templates"}
     3. Validate content_type: must be "slash_commands" or "agent_templates"
     4. Call TokenManager.generate_token()
     5. Stage files to temp/{tenant_key}/{token}/ directory
     6. Mark token as "ready" (staging_status="ready")
     7. Return download URL: http://server:7272/api/download/temp/{token}/agent-templates.zip

     **Step 2: Download File** (GET /api/download/temp/{token}/{filename})
     File: api/endpoints/downloads.py (lines 595-699)

     1. NO AUTHENTICATION REQUIRED (public endpoint - token IS the auth)
     2. Validate token via TokenManager.validate_token()
     3. Read file from temp/{tenant_key}/{token}/{filename}
     4. Increment download_count (unlimited downloads within 15 min window)
     5. Serve ZIP file

     from datetime import datetime

     class Contact(db.Model):
         """Contact model for storing contact information"""

         __tablename__ = 'contacts'

         # Primary key
         id = db.Column(db.Integer, primary_key=True)

         # Basic information
         first_name = db.Column(db.String(50), nullable=False)
         last_name = db.Column(db.String(50), nullable=False)
         email = db.Column(db.String(120), unique=True, nullable=True)
         phone = db.Column(db.String(20), nullable=True)

         # Additional information
         company = db.Column(db.String(100), nullable=True)
         job_title = db.Column(db.String(100), nullable=True)
         notes = db.Column(db.Text, nullable=True)

         # Profile picture
         profile_picture = db.Column(db.String(200), nullable=True)  # File path

         # Timestamps
         created_at = db.Column(db.DateTime, default=datetime.utcnow)
         updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

         def __repr__(self):
             return f'<Contact {self.first_name} {self.last_name}>'

         @property
         def full_name(self):
             """Return the full name of the contact"""
             return f"{self.first_name} {self.last_name}"

         def to_dict(self):
             """Convert contact to dictionary for JSON serialization"""
             return {
                 'id': self.id,
                 'first_name': self.first_name,
                 'last_name': self.last_name,
                 'email': self.email,
                 'phone': self.phone,
                 'company': self.company,
                 'job_title': self.job_title,
                 'notes': self.notes,
                 'profile_picture': self.profile_picture,
                 'created_at': self.created_at.isoformat() if self.created_at else None,
                 'updated_at': self.updated_at.isoformat() if self.updated_at else None

  ⎿  # TinyContacts

     A lightweight, modern contact management application built with Flask and SQLAlchemy.

     ## Overview

     TinyContacts is a simple yet powerful contact management system that allows users to store, organize, and manage
     their personal and professional contacts. The application focuses on essential contact management features with a
     clean, intuitive interface.

     ## Features

     ### Core Contact Management
     - **CRUD Operations**: Create, read, update, and delete contacts
     - **Contact Information**: Store names, phone numbers, emails, addresses, and notes
     - **Photo Upload**: Add and manage contact photos
     - **Date Management**: Track important dates like birthdays and anniversaries
     - **Search and Filter**: Quickly find contacts using various criteria

     ### User Experience
     - **Responsive Design**: Works seamlessly across desktop and mobile devices
     - **Clean Interface**: Minimalist design focused on usability
     - **Quick Actions**: One-click access to common operations
     - **Data Export**: Export contact data for backup and portability

     ## Technology Stack

     ### Backend
     - **Flask**: Lightweight Python web framework
     - **SQLAlchemy**: Object-relational mapping (ORM)
     - **Flask-Migrate**: Database migration management
     - **Flask-WTF**: Form handling and validation
     - **Werkzeug**: File upload handling

     ### Frontend
     - **Jinja2**: Template engine for dynamic HTML
     - **Bootstrap**: Responsive CSS framework
     - **JavaScript**: Interactive frontend features

     ### Database
     - **SQLite**: Default development database
     - **PostgreSQL**: Production database option

     ### File Storage
     - **Local Storage**: Photo and file management
     - **Configurable Paths**: Customizable upload directories

     # Search contacts by name
